192 void
193 SUMOVTypeParameter::write(OutputDevice& dev) const {
194  if (onlyReferenced) {
195  return;
196  }
197  dev.openTag(SUMO_TAG_VTYPE);
198  dev.writeAttr(SUMO_ATTR_ID, id);
199  if (wasSet(VTYPEPARS_LENGTH_SET)) {
200  dev.writeAttr(SUMO_ATTR_LENGTH, length);
201  }
202  if (wasSet(VTYPEPARS_MINGAP_SET)) {
203  dev.writeAttr(SUMO_ATTR_MINGAP, minGap);
204  }
205  if (wasSet(VTYPEPARS_MAXSPEED_SET)) {
206  dev.writeAttr(SUMO_ATTR_MAXSPEED, maxSpeed);
207  }
208  if (wasSet(VTYPEPARS_PROBABILITY_SET)) {
209  dev.writeAttr(SUMO_ATTR_PROB, defaultProbability);
210  }
211  if (wasSet(VTYPEPARS_SPEEDFACTOR_SET)) {
212  dev.writeAttr(SUMO_ATTR_SPEEDFACTOR, speedFactor);
213  }
214  if (wasSet(VTYPEPARS_SPEEDDEVIATION_SET)) {
215  dev.writeAttr(SUMO_ATTR_SPEEDDEV, speedDev);
216  }
217  if (wasSet(VTYPEPARS_VEHICLECLASS_SET)) {
218  dev.writeAttr(SUMO_ATTR_VCLASS, toString(vehicleClass));
219  }
220  if (wasSet(VTYPEPARS_EMISSIONCLASS_SET)) {
221  dev.writeAttr(SUMO_ATTR_EMISSIONCLASS, PollutantsInterface::getName(emissionClass));
222  }
223  if (wasSet(VTYPEPARS_IMPATIENCE_SET)) {
224  if (impatience == -std::numeric_limits<SUMOReal>::max()) {
225  dev.writeAttr(SUMO_ATTR_IMPATIENCE, "off");
226  } else {
227  dev.writeAttr(SUMO_ATTR_IMPATIENCE, impatience);
228  }
229  }
230  if (wasSet(VTYPEPARS_SHAPE_SET)) {
231  dev.writeAttr(SUMO_ATTR_GUISHAPE, getVehicleShapeName(shape));
232  }
233  if (wasSet(VTYPEPARS_WIDTH_SET)) {
234  dev.writeAttr(SUMO_ATTR_WIDTH, width);
235  }
236  if (wasSet(VTYPEPARS_HEIGHT_SET)) {
237  dev.writeAttr(SUMO_ATTR_HEIGHT, height);
238  }
239  if (wasSet(VTYPEPARS_COLOR_SET)) {
240  dev.writeAttr(SUMO_ATTR_COLOR, color);
241  }
242  if (wasSet(VTYPEPARS_OSGFILE_SET)) {
243  dev.writeAttr(SUMO_ATTR_OSGFILE, osgFile);
244  }
245  if (wasSet(VTYPEPARS_IMGFILE_SET)) {
246  dev.writeAttr(SUMO_ATTR_IMGFILE, imgFile);
247  }
248  if (wasSet(VTYPEPARS_LANE_CHANGE_MODEL_SET)) {
249  dev.writeAttr(SUMO_ATTR_LANE_CHANGE_MODEL, lcModel);
250  }
251  if (wasSet(VTYPEPARS_PERSON_CAPACITY)) {
252  dev.writeAttr(SUMO_ATTR_PERSON_CAPACITY, personCapacity);
253  }
254  if (wasSet(VTYPEPARS_CONTAINER_CAPACITY)) {
255  dev.writeAttr(SUMO_ATTR_CONTAINER_CAPACITY, containerCapacity);
256  }
257  if (wasSet(VTYPEPARS_BOARDING_DURATION)) {
258  dev.writeAttr(SUMO_ATTR_BOARDING_DURATION, boardingDuration);
259  }
260  if (wasSet(VTYPEPARS_LOADING_DURATION)) {
261  dev.writeAttr(SUMO_ATTR_LOADING_DURATION, loadingDuration);
262  }
263 
264  if (cfParameter.size() != 0) {
265  dev.openTag(cfModel);
266  std::vector<SumoXMLAttr> attrs;
267  for (CFParams::const_iterator i = cfParameter.begin(); i != cfParameter.end(); ++i) {
268  attrs.push_back(i->first);
269  }
270  std::sort(attrs.begin(), attrs.end());
271  for (std::vector<SumoXMLAttr>::const_iterator i = attrs.begin(); i != attrs.end(); ++i) {
272  dev.writeAttr(*i, cfParameter.find(*i)->second);
273  }
274  dev.closeTag();
275  }
276  for (std::map<std::string, std::string>::const_iterator j = getMap().begin(); j != getMap().end(); ++j) {
277  dev.openTag(SUMO_TAG_PARAM);
278  dev.writeAttr(SUMO_ATTR_KEY, (*j).first);
279  dev.writeAttr(SUMO_ATTR_VALUE, (*j).second);
280  dev.closeTag();
281  }
282  dev.closeTag();
283 }
