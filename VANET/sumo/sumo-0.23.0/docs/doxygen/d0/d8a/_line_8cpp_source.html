100 Position
101 Line::getPositionAtDistance(SUMOReal offset) const {
102  const SUMOReal length = myP1.distanceTo(myP2);
103  if (length == 0) {
104  if (offset != 0) {
105  throw InvalidArgument("Invalid offset " + toString(offset) + " for Line with length " + toString(length));;
106  }
107  return myP1;
108  }
109  return myP1 + (myP2 - myP1) * (offset / length);
110 }
