150 int
151 NBContHelper::relative_outgoing_edge_sorter::operator()(NBEdge* e1, NBEdge* e2) const {
152  if (e1 == 0 || e2 == 0) {
153  return -1;
154  }
155  SUMOReal relAngle1 = NBHelpers::normRelAngle(
156  myEdge->getEndAngle(), e1->getStartAngle());
157  SUMOReal relAngle2 = NBHelpers::normRelAngle(
158  myEdge->getEndAngle(), e2->getStartAngle());
159 
160  SUMOReal lookAhead = 2 * NBEdge::ANGLE_LOOKAHEAD;
161  while (fabs(relAngle1 - relAngle2) < 3.0) {
162  // look at further geometry segments to resolve ambiguity
163  const Position referencePos1 = e1->getGeometry().positionAtOffset2D(lookAhead);
164  const Position referencePos2 = e2->getGeometry().positionAtOffset2D(lookAhead);
165  relAngle1 = NBHelpers::normRelAngle(myEdge->getEndAngle(), NBHelpers::angle(
166  e1->getFromNode()->getPosition().x(), e1->getFromNode()->getPosition().y(),
167  referencePos1.x(), referencePos1.y()));
168  relAngle2 = NBHelpers::normRelAngle(myEdge->getEndAngle(), NBHelpers::angle(
169  e2->getFromNode()->getPosition().x(), e2->getFromNode()->getPosition().y(),
170  referencePos2.x(), referencePos2.y()));
171  if (lookAhead > MAX2(e1->getLength(), e2->getLength())) {
172  break;
173  }
174  lookAhead *= 2;
175  }
176  return relAngle1 > relAngle2;
177 }

184 NBContHelper::relative_incoming_edge_sorter::operator()(NBEdge* e1, NBEdge* e2) const {
185  if (e1 == 0 || e2 == 0) {
186  return -1;
187  }
188  SUMOReal relAngle1 = NBHelpers::normRelAngle(
189  myEdge->getStartAngle(), e1->getEndAngle());
190  SUMOReal relAngle2 = NBHelpers::normRelAngle(
191  myEdge->getStartAngle(), e2->getEndAngle());
192 
193  SUMOReal lookAhead = 2 * NBEdge::ANGLE_LOOKAHEAD;
194  while (fabs(relAngle1 - relAngle2) < 3.0) {
195  // look at further geometry segments to resolve ambiguity
196  const Position referencePos1 = e1->getGeometry().positionAtOffset2D(e1->getGeometry().length() - lookAhead);
197  const Position referencePos2 = e2->getGeometry().positionAtOffset2D(e2->getGeometry().length() - lookAhead);
198  relAngle1 = NBHelpers::normRelAngle(myEdge->getStartAngle(), NBHelpers::angle(
199  referencePos1.x(), referencePos1.y(),
200  e1->getToNode()->getPosition().x(), e1->getToNode()->getPosition().y()));
201  relAngle2 = NBHelpers::normRelAngle(myEdge->getStartAngle(), NBHelpers::angle(
202  referencePos2.x(), referencePos2.y(),
203  e2->getToNode()->getPosition().x(), e2->getToNode()->getPosition().y()));
204  if (lookAhead > MAX2(e1->getLength(), e2->getLength())) {
205  break;
206  }
207  lookAhead *= 2;
208  }
209  return relAngle1 > relAngle2;
210 }

257 int
258 NBContHelper::edge_by_angle_to_nodeShapeCentroid_sorter::operator()(const NBEdge* e1, const NBEdge* e2) const {
259  assert(e1->getFromNode() == myNode || e1->getToNode() == myNode);
260  assert(e2->getFromNode() == myNode || e2->getToNode() == myNode);
261  const SUMOReal angle1 = e1->getAngleAtNodeToCenter(myNode);
262  const SUMOReal angle2 = e2->getAngleAtNodeToCenter(myNode);
263  const SUMOReal absDiff = abs(angle1 - angle2);
264 
265  // cannot trust the angle difference hence a heuristic:
266  if (absDiff < 2 || absDiff > (360 - 2)) {
267  const bool sameDir = ((e1->getFromNode() == myNode && e2->getFromNode() == myNode)
268  || (e1->getToNode() == myNode && e2->getToNode() == myNode));
269  if (sameDir) {
270  // put edges that allow pedestrians on the 'outside'
271  if (e1->getToNode() == myNode) {
272  return (e1->getPermissions() & SVC_PEDESTRIAN) != 0;
273  } else {
274  return (e1->getPermissions() & SVC_PEDESTRIAN) == 0;
275  }
276  } else {
277  // sort incoming before outgoing
278  return e1->getToNode() == myNode;
279  }
280  }
281  return angle1 < angle2;
282 }
