106 GUIInductLoop::MyWrapper::MyWrapper(GUIInductLoop& detector, SUMOReal pos)
107  : GUIDetectorWrapper("induct loop", detector.getID()),
108  myDetector(detector), myPosition(pos) {
109  myFGPosition = detector.getLane()->geometryPositionAtOffset(pos);

111  myBoundary.add(myFGPosition.x() - (SUMOReal) 5.5, myFGPosition.y() - (SUMOReal) 5.5);
112  myFGRotation = -detector.getLane()->getShape().rotationDegreeAtOffset(pos);
113 }

153 void
154 GUIInductLoop::MyWrapper::drawGL(const GUIVisualizationSettings& s) const {
155  glPushName(getGlID());
156  SUMOReal width = (SUMOReal) 2.0 * s.scale;
157  glLineWidth(1.0);
158  const SUMOReal exaggeration = s.addSize.getExaggeration(s);
159  // shape
160  glColor3d(1, 1, 0);
161  glPushMatrix();
162  glTranslated(0, 0, getType());
163  glTranslated(myFGPosition.x(), myFGPosition.y(), 0);
164  glRotated(myFGRotation, 0, 0, 1);
165  glScaled(exaggeration, exaggeration, 1);
166  glBegin(GL_QUADS);
167  glVertex2d(0 - 1.0, 2);
168  glVertex2d(-1.0, -2);
169  glVertex2d(1.0, -2);
170  glVertex2d(1.0, 2);
171  glEnd();
172  glTranslated(0, 0, .01);
173  glBegin(GL_LINES);
174  glVertex2d(0, 2 - .1);
175  glVertex2d(0, -2 + .1);
176  glEnd();
177 
178  // outline
179  if (width * exaggeration > 1) {
180  glColor3d(1, 1, 1);
181  glPolygonMode(GL_FRONT_AND_BACK, GL_LINE);
182  glBegin(GL_QUADS);
183  glVertex2f(0 - 1.0, 2);
184  glVertex2f(-1.0, -2);
185  glVertex2f(1.0, -2);
186  glVertex2f(1.0, 2);
187  glEnd();
188  glPolygonMode(GL_FRONT_AND_BACK, GL_FILL);
189  }
190 
191  // position indicator
192  if (width * exaggeration > 1) {
193  glRotated(90, 0, 0, -1);
194  glColor3d(1, 1, 1);
195  glBegin(GL_LINES);
196  glVertex2d(0, 1.7);
197  glVertex2d(0, -1.7);
198  glEnd();
199  }
200  glPopMatrix();
201  drawName(getCenteringBoundary().getCenter(), s.scale, s.addName);
202  glPopName();
203 }
